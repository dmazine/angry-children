## Installation

```sh
$ npm install --save angry-children
```

## Usage

```js
const maxMin = require('angry-children');

const k = 3;
const arr = [10, 100, 300, 200, 1000, 20, 30];

let minimumUnfairness = maxMin(k, arr);
```
